  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Edit Sections: ${widget.page.title}'),
        backgroundColor: Colors.white,
        foregroundColor: Colors.black,
        elevation: 1,
        actions: [
          IconButton(
            icon: const Icon(Icons.add),
            onPressed: _addSection,
            tooltip: 'Add Section',
          ),
        ],
      ),
      body: _isLoading
          ? const Center(child: CircularProgressIndicator())
          : _sections.isEmpty
              ? Center(
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      Icon(Icons.view_agenda_outlined,
                          size: 64, color: Colors.grey[400]),
                      const SizedBox(height: 16),
                      const Text('No sections yet'),
                      const SizedBox(height: 16),
                      ElevatedButton.icon(
                        onPressed: _addSection,
                        icon: const Icon(Icons.add),
                        label: const Text('Add First Section'),
                      ),
                    ],
                  ),
                )
              : ReorderableListView.builder(
                  padding: const EdgeInsets.all(16),
                  itemCount: _sections.length,
                  onReorder: _reorderSections,
                  itemBuilder: (context, index) {
                    final section = _sections[index];
                    return Card(
                      key: ValueKey(section.id),
                      margin: const EdgeInsets.only(bottom: 12),
                      child: ListTile(
                        leading: Row(
                          mainAxisSize: MainAxisSize.min,
                          children: [
                            ReorderableDragStartListener(
                              index: index,
                              child: const Icon(Icons.drag_handle,
                                  color: Colors.grey),
                            ),
                            const SizedBox(width: 12),
                            Container(
                              width: 40,
                              height: 40,
                              decoration: BoxDecoration(
                                color: Colors.blue.shade50,
                                borderRadius: BorderRadius.circular(8),
                              ),
                              child: Icon(
                                _getSectionIcon(section.type),
                                color: Colors.blue.shade700,
                              ),
                            ),
                          ],
                        ),
                        title: Text(
                          (section.title?.isNotEmpty ?? false)
                              ? section.title!
                              : section.type,
                          style: const TextStyle(fontWeight: FontWeight.w600),
                        ),
                        subtitle: Row(
                          children: [
                            Text(section.type),
                            const SizedBox(width: 8),
                            Container(
                              padding: const EdgeInsets.symmetric(
                                  horizontal: 6, vertical: 2),
                              decoration: BoxDecoration(
                                color: section.isActive
                                    ? Colors.green.shade50
                                    : Colors.grey.shade200,
                                borderRadius: BorderRadius.circular(4),
                              ),
                              child: Text(
                                section.isActive ? 'Active' : 'Inactive',
                                style: TextStyle(
                                    fontSize: 10,
                                    color: section.isActive
                                        ? Colors.green.shade700
                                        : Colors.grey.shade600),
                              ),
                            ),
                          ],
                        ),
                        trailing: Row(
                          mainAxisSize: MainAxisSize.min,
                          children: [
                            // Reorder buttons
                            Column(
                              mainAxisSize: MainAxisSize.min,
                              children: [
                                InkWell(
                                  onTap: index > 0
                                      ? () => _moveSectionUp(index)
                                      : null,
                                  child: Icon(
                                    Icons.arrow_drop_up,
                                    color: index > 0
                                        ? Colors.blue.shade700
                                        : Colors.grey.shade300,
                                    size: 24,
                                  ),
                                ),
                                InkWell(
                                  onTap: index < _sections.length - 1
                                      ? () => _moveSectionDown(index)
                                      : null,
                                  child: Icon(
                                    Icons.arrow_drop_down,
                                    color: index < _sections.length - 1
                                        ? Colors.blue.shade700
                                        : Colors.grey.shade300,
                                    size: 24,
                                  ),
                                ),
                              ],
                            ),
                            const SizedBox(width: 8),
                            IconButton(
                              icon: const Icon(Icons.edit),
                              onPressed: () => _editSection(section, index),
                              tooltip: 'Edit Section',
                            ),
                            IconButton(
                              icon: const Icon(Icons.delete, color: Colors.red),
                              onPressed: () => _deleteSection(section, index),
                              tooltip: 'Delete Section',
                            ),
                          ],
                        ),
                      ),
                    );
                  },
                ),
      floatingActionButton: FloatingActionButton(
        onPressed: _addSection,
        backgroundColor: Colors.black,
        child: const Icon(Icons.add),
      ),
    );
  }
